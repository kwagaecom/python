# Simple To-Do List Application # Initialize an empty list to store tasks tasks = [] def display_menu(): print("\nTo-Do List Application") print("1. View Tasks") print("2. Add Task") print("3. Remove Task") print("4. Exit") def view_tasks(): if not tasks: print("\nYour task list is empty!") else: print("\nYour Tasks:") for i, task in enumerate(tasks, start=1): print(f"{i}. {task}") def add_task(): task = input("Enter the task: ") tasks.append(task) print(f'"{task}" has been added to your list.') def remove_task(): view_tasks() if tasks: try: task_num = int(input("Enter the task number to remove: ")) if 1 <= task_num <= len(tasks): removed_task = tasks.pop(task_num - 1) print(f'"{removed_task}" has been removed from your list.') else: print("Invalid task number.") except ValueError: print("Please enter a valid number.") def main(): while True: display_menu() choice = input("Choose an option (1-4): ") if choice == "1": view_tasks() elif choice == "2": add_task() elif choice == "3": remove_task() elif choice == "4": print("Exiting the application. Goodbye!") break else: print("Invalid choice. Please try again.") if __name__ == "__main__": main()
